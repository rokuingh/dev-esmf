\begin{verbatim}
use ESMF_Mod

! Declare a Calendar
type(ESMF_Calendar) :: gregorian

! Set it to be a Gregorian calendar
gregorian = ESMF_CalendarCreate("Gregorian", ESMF_CAL_GREGORIAN)

! pass it to a time instant
call ESMF_TimeSet(startTime, yy=2002, mm=8, dd=1, gregorian)
\end{verbatim}
